23 static inline void G(int r, int i, uint64_t& a, uint64_t& b, uint64_t& c, uint64_t& d, uint64_t* M)
24 {
25 a = a + b + M[blake2_constants<void>::S[r][2 * i]];
26 d = rotate_right(d ^ a, 32);
27 c = c + d;
28 b = rotate_right(b ^ c, 24);
29 a = a + b + M[blake2_constants<void>::S[r][2 * i + 1]];
30 d = rotate_right(d ^ a, 16);
31 c = c + d;
32 b = rotate_right(b ^ c, 63);
33 }

148 inline void init()
149 {
150 pos = 0;
151 total = 0;
152 xoffset = 0;
153 squeezing = false;
154
155 blake2_functions::initH(H);
156
157 if (type == blake2_type::hash)
158 H[0] ^= hash_size()/8;
159 else
160 {
161 H[0] ^= N / 8;
162 T rhs = type == blake2_type::x_hash ? static_cast<T>(hs/8) : (static_cast<T>(-1) >> (sizeof(T) * 4));
163 if (N == 512)
164 H[1] ^= (rhs << (N / 16));
165 else
166 H[3] ^= rhs;
167 }
168 H[0] ^= (k.size() << 8);
169 H[4] ^= s[0];
170 H[5] ^= s[1];
171 H[6] ^= p[0];
172 H[7] ^= p[1];
173 absorb_key();
174 }

206 inline void squeeze(unsigned char* hash, size_t hs)
207 {
208 size_t processed = 0;
209 if (!squeezing)
210 {
211 if (type == blake2_type::xof)
212 total += pos * 8;
213 squeezing = true;
214 xoffset = 0;
215 if (N / 4 != pos)
216 memset(&m[pos], 0, N / 4 - pos);
217 transform(m.data(), 1, true);
218 memcpy(&m[0], H.data(), N / 8);
219 }
220 else if (pos < N / 8)
221 {
222 size_t to_copy = std::min(hs, N / 8 - pos);
223 memcpy(hash, reinterpret_cast<unsigned char*>(H.data()) + pos, to_copy);
224 processed += to_copy;
225 pos += to_copy;
226 }
227 while (processed < hs)
228 {
229 blake2_functions::initX(H, hs, processed, xoffset++, type == blake2_type::xof ? static_cast<T>(-1) : static_cast<T>(hs));
230 H[4] ^= s[0];
231 H[5] ^= s[1];
232 H[6] ^= p[0];
233 H[7] ^= p[1];
234 total = N;
235 memset(&m[N / 8], 0, m.size() - N / 8);
236 transform(m.data(), 1, true);
237 pos = std::min(hs - processed, N / 8);
238 memcpy(hash + processed, H.data(), pos);
239 processed += pos;
240 }
241 }

243 inline void final(unsigned char* hash)
244 {
245 total += pos * 8;
246 if (type == blake2_type::hash)
247 {
248 if (N / 4 != pos)
249 memset(&m[pos], 0, N / 4 - pos);
250 transform(m.data(), 1, true);
251 memcpy(hash, H.data(), hash_size() / 8);
252 }
253 else
254 squeeze(hash, hs / 8);
255 }

282 inline void transform(const unsigned char* data, size_t num_blks, bool padding)
283 {
284 for (size_t blk = 0; blk < num_blks; blk++)
285 {
286 T M[16];
287 for (int i = 0; i < 16; i++)
288 M[i] = reinterpret_cast<const T*>(data)[blk * 16 + i];
289 uint64_t totalbytes = total / 8 + (padding ? 0 : (blk + 1) * N) / 4;
290 T t0 = static_cast<T>(totalbytes);
291 T t1 = N == 512 ? 0 : static_cast<T>(totalbytes >> 32);
292 T f0 = 0;
293 T f1 = 0;
294 if (padding)
295 {
296 f0 = static_cast<T>(-1);
297 f1 = 0;
298 }
299
300 T v[16];
301 memcpy(v, H.data(), sizeof(T) * 8);
302 v[8 + 0] = blake2_functions::IV<T>(0);
303 v[8 + 1] = blake2_functions::IV<T>(1);
304 v[8 + 2] = blake2_functions::IV<T>(2);
305 v[8 + 3] = blake2_functions::IV<T>(3);
306 v[12] = t0 ^ blake2_functions::IV<T>(4);
307 v[13] = t1 ^ blake2_functions::IV<T>(5);
308 v[14] = f0 ^ blake2_functions::IV<T>(6);
309 v[15] = f1 ^ blake2_functions::IV<T>(7);
310
311 blake2_functions::round(0, M, v);
312 blake2_functions::round(1, M, v);
313 blake2_functions::round(2, M, v);
314 blake2_functions::round(3, M, v);
315 blake2_functions::round(4, M, v);
316 blake2_functions::round(5, M, v);
317 blake2_functions::round(6, M, v);
318 blake2_functions::round(7, M, v);
319 blake2_functions::round(8, M, v);
320 blake2_functions::round(9, M, v);
321 if (N == 512)
322 {
323 blake2_functions::round(10, M, v);
324 blake2_functions::round(11, M, v);
325 }
326
327 H[0] = H[0] ^ v[0] ^ v[0 + 8];
328 H[0 + 4] = H[0 + 4] ^ v[0 + 4] ^ v[0 + 8 + 4];
329 H[1] = H[1] ^ v[1] ^ v[1 + 8];
330 H[1 + 4] = H[1 + 4] ^ v[1 + 4] ^ v[1 + 8 + 4];
331 H[2] = H[2] ^ v[2] ^ v[2 + 8];
332 H[2 + 4] = H[2 + 4] ^ v[2 + 4] ^ v[2 + 8 + 4];
333 H[3] = H[3] ^ v[3] ^ v[3 + 8];
334 H[3 + 4] = H[3 + 4] ^ v[3 + 4] ^ v[3 + 8 + 4];
335 }
336
337 }
